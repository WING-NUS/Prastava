 def read_config_file
        config_hash = {}
        File.open($HOME+"/client/config.ini") do |file|
                 while !file.eof?
                        line = file.readline
                                if line =~ /^#/ or line == "\n"
                                       next
                                end
                        (parameter, value) = line.split("=")
                        config_hash[parameter] = value.chomp
                end
        end
        config_hash
  end
