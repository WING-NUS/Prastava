  def to_terms (my_string)
        terms_list = my_string.downcase.gsub(/(\s|\d|\W)+/u,' ').rstrip.strip.split(' ').reject{|term|$stop_words.include?(term)}#.each {|term| term = term.stem}
        count = 0
        string = ""
        terms_list.each do|term|
                count+=1
                term = term.stem
                string = string + " " + term
        end
        string
  end
